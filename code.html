Youth Valley Hackathon (Java)

Example: win-check logic (tic-tac-toe).

// Java — example snippet
static boolean hasWinner(char[][] b, char p) {
    for (int i = 0; i < 3; i++) {
        if (b[i][0] == p && b[i][1] == p && b[i][2] == p) return true;
        if (b[0][i] == p && b[1][i] == p && b[2][i] == p) return true;
    }
    return (b[0][0] == p && b[1][1] == p && b[2][2] == p)
        || (b[0][2] == p && b[1][1] == p && b[2][0] == p);
}

// C — example snippet
static inline unsigned set_bit(unsigned x, unsigned bit) {
    return x | (1u << bit);
}

static inline unsigned clear_bit(unsigned x, unsigned bit) {
    return x & ~(1u << bit);
}

// Java — example snippet
Object eval(Expression expr) {
    return expr.accept(new ExpressionVisitor<Object>() {
        public Object visitNumber(NumberExpr n) { return n.value(); }
        public Object visitBinary(BinaryExpr b) {
            Object l = eval(b.left());
            Object r = eval(b.right());
            return ((double) l) + ((double) r); // placeholder
        }
    });
